# -------------------------------------------
# Application Settings
# -------------------------------------------
# APP_NAME: Used in logs, health checks, and API metadata.
# APP_ENV: Controls debug mode, error verbosity, and feature flags.
#   Options: development | staging | production
# APP_PORT: The port the FastAPI server listens on.
APP_NAME=restaurant-booking-api
APP_ENV=development
APP_PORT=8000

# -------------------------------------------
# Database Configuration
# -------------------------------------------
# DATABASE_URL: Full connection string for PostgreSQL.
#   Format: postgresql://<user>:<password>@<host>:<port>/<database>
#   For Azure Database for PostgreSQL, enable SSL by appending ?sslmode=require
#   Example: postgresql://admin:secret@mydb.postgres.database.azure.com:5432/restaurant_db?sslmode=require
DATABASE_URL=postgresql://user:password@host:5432/restaurant_db

# -------------------------------------------
# JWT Authentication
# -------------------------------------------
# JWT_SECRET_KEY: Secret used to sign and verify tokens.
#   IMPORTANT: Use a strong, random string in production (min 32 characters).
#   Generate one with: python -c "import secrets; print(secrets.token_urlsafe(64))"
# JWT_ALGORITHM: Signing algorithm. HS256 is standard for symmetric keys.
# JWT_EXPIRY_MINUTES: How long access tokens remain valid before requiring re-auth.
JWT_SECRET_KEY=your-secret-key-here-change-in-production
JWT_ALGORITHM=HS256
JWT_EXPIRY_MINUTES=30

# -------------------------------------------
# Timezone Configuration
# -------------------------------------------
# TIMEZONE_OFFSET: UTC offset in hours for all system timestamps.
#   This affects JWT expiry calculations, booking created_at/updated_at,
#   and any time-based conflict detection logic.
#   Example: 7 = UTC+7 (Western Indonesia / Bangkok / Ho Chi Minh City)
TIMEZONE_OFFSET=7

# -------------------------------------------
# SMS Notification Service (smsnotif.id)
# -------------------------------------------
# SMS_API_URL: Base URL for the SMS gateway API.
# SMS_API_KEY: Your API key from smsnotif.id dashboard.
#   Request a quota increase if you expect high booking volume.
# SMS_SENDER_ID: The sender name that appears on the recipient's phone.
#   Must be registered/approved with your SMS provider.
SMS_API_URL=https://api.smsnotif.id
SMS_API_KEY=your-sms-api-key-here
SMS_SENDER_ID=RestaurantApp

# -------------------------------------------
# Azure Key Vault (Optional - Production Only)
# -------------------------------------------
# When APP_ENV=production, secrets like JWT_SECRET_KEY, DATABASE_URL,
# and SMS_API_KEY should be fetched from Azure Key Vault instead of
# this file. Set the vault URL below and ensure your Azure identity
# (Managed Identity or Service Principal) has "Get Secret" permissions.
# Leave empty or remove in development/staging.
AZURE_KEY_VAULT_URL=https://your-vault.vault.azure.net/

# -------------------------------------------
# Logging
# -------------------------------------------
# LOG_LEVEL: Controls verbosity of application logs.
#   Options: DEBUG | INFO | WARNING | ERROR | CRITICAL
#   Use DEBUG in development, INFO or WARNING in production.
LOG_LEVEL=INFO

# -------------------------------------------
# CORS (Cross-Origin Resource Sharing)
# -------------------------------------------
# CORS_ORIGINS: Comma-separated list of allowed frontend origins.
#   These are the URLs where your frontend app runs.
#   In production, restrict this to your actual domain(s).
#   Example: https://restaurant.example.com
CORS_ORIGINS=http://localhost:3000,http://localhost:8080
